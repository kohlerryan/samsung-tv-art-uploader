# ==============================
# Radxa First Boot Configuration
# ==============================
#
# This file uses the Radxa first-boot DSL — it is NOT a bash script.
# Keep this file as-is. The samsung-tv-art Docker setup is done separately
# by running setup.sh after SSH-ing in (see examples/headless/setup.sh).
#
# ── Before using ─────────────────────────────────────────────────────────────
#   1. Change the password in the add_user line below (currently "changeme")
#   2. Update the codenames in update_generic_hostname to match your image
#      (check the default before.txt on the flashed image for the correct values)
#   3. To set a custom hostname after first boot:
#        sudo hostnamectl set-hostname samsung-tv-art
#
#   NOTE: This file uses a simple DSL — shell variables are NOT supported.
#         Edit values directly in the commands below.
# ─────────────────────────────────────────────────────────────────────────────

# Allow config to continue even when some commands fail
no_fail
log "Running before.txt as first boot configuration"

# Update generic hostname — replace codenames with those from your image's default before.txt
update_generic_hostname radxa-zero3

# Create default accounts — change "changeme" to your desired password
add_user uploader changeme
user_append_group uploader sudo
user_append_group uploader audio
user_append_group uploader video
user_append_group uploader plugdev
user_append_group uploader render
user_append_group uploader gpio
user_append_group uploader i2c
user_append_group uploader spidev
user_append_group uploader pwm

# Resize root partition at the filesystem level
resize_root

# Disable services
disable_service smbd
disable_service nmbd
disable_service systemd-networkd

# Generate unique hardware fingerprint
regenerate_ssh_hostkey

# Configure locale
update_locale en_US.UTF-8

# ── Network ───────────────────────────────────────────────────────────────────
# Option A: Ethernet (USB-C adapter or dock) — no extra config needed.
#
# Option B: WiFi only — uncomment and fill in your credentials.
#   The board will be reachable over SSH on first power-on without a monitor.
#
# connect_wi-fi "YourSSID" "YourPassword"

# Enable SSH (always — this is a headless server)
enable_service ssh

# Remove first-boot package
remove_packages rsetup-config-first-boot
